.. py:class:: GenericCodeBlockExtractor(file_content: str, file_name: str)

   Extracts code blocks from source files based on detected programming language.

   :param file_content: The content of the source file.
   :type file_content: str
   :param file_name: The name of the source file, used for language detection.
   :type file_name: str

   :returns: A list of extracted code blocks with headers and footers.
   :rtype: List[str]


   .. py:attribute:: LANGUAGE_PATTERNS


   .. py:attribute:: EXT_LANGUAGE_MAP


   .. py:attribute:: file_content


   .. py:attribute:: file_name


   .. py:attribute:: language
      :value: 'python'



   .. py:method:: _detect_language()

      Detects the programming language based on the file extension.

          Returns:
              str: The detected language or 'python' as default.




   .. py:method:: code_block_extractor() -> List[str]

      Extracts all code blocks (including nested) for the detected language.



   .. py:method:: _extract_blocks_recursive(lines: List[str], start: int, end: int, blocks: List[str], base_indent: int)

      Recursively extracts code blocks from a list of lines based on language-specific patterns.

      :param lines: The lines of code to analyze.
      :type lines: List[str]
      :param start: The starting index for extraction.
      :type start: int
      :param end: The ending index for extraction.
      :type end: int
      :param blocks: The list to store extracted blocks.
      :type blocks: List[str]
      :param base_indent: The base indentation level for the blocks.
      :type base_indent: int

      :returns: None



   .. py:method:: _extract_single_block(lines: List[str], start_idx: int, pattern: str) -> dict
